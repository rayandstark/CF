串接 Google Sheets
設定完成後，每次在此新增交易或固定支出，都會自動同步寫入你的 Google Sheets，資料永久保存。
1
開啟 Google Sheets，建立新試算表，命名為「現金流追蹤」
2
點選選單 擴充功能 → Apps Script，將下方程式碼完整貼入，按 Ctrl+S 儲存
3
點 部署 → 新增部署作業，類型選「網路應用程式」，執行身分選「我」，存取權選「任何人」，按「部署」
4
授權後複製產生的「網路應用程式 URL」，貼到下方欄位並按「儲存並測試」
5
之後每筆新增的交易都會自動出現在 Sheets 的「交易」工作表
Apps Script 程式碼（貼入 Apps Script 編輯器）
function doPost(e) { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.type === 'tx') { const sheet = ss.getSheetByName('交易') || ss.insertSheet('交易'); if (sheet.getLastRow() === 0) { sheet.appendRow(['日期', '項目', '分類', '類型', '金額', '備註', '同步時間']); sheet.getRange(1,1,1,7).setFontWeight('bold'); } sheet.appendRow([ data.date, data.name, data.cat, data.txType === 'income' ? '收入' : '支出', data.amount, data.note || '', new Date().toLocaleString('zh-TW') ]); } if (data.type === 'fixed') { const sheet = ss.getSheetByName('固定支出') || ss.insertSheet('固定支出'); if (sheet.getLastRow() === 0) { sheet.appendRow(['項目', '分類', '每月金額', '扣款日', '同步時間']); sheet.getRange(1,1,1,5).setFontWeight('bold'); } sheet.appendRow([ data.name, data.cat, data.amount, data.day || '未設定', new Date().toLocaleString('zh-TW') ]); } return ContentService .createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ ok: true, msg: 'connected' })) .setMimeType(ContentService.MimeType.JSON); }
